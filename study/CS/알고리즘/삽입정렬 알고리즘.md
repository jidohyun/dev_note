```c
#include <stdio.h>

#define LEN 5 // 배열의 크기 정의

void insertionSort(int* arr) {
    int i, j, key;
    for (i = 1; i < LEN; i++) {
        key = arr[i]; // 현재 삽입될 숫자
        for (j = i - 1; j >= 0 && arr[j] > key; j--) { // key보다 큰 값 이동
            arr[j + 1] = arr[j]; // 한 칸 뒤로 이동
        }
        arr[j + 1] = key; // 알맞은 위치에 key 삽입
    }
}

int main() {
    int arr[LEN] = {64, 25, 12, 22, 11}; // 정렬할 배열 초기화
    insertionSort(arr); // 삽입 정렬 호출

    printf("정렬된 배열: ");
    for (int i = 0; i < LEN; i++) {
        printf("%d ", arr[i]); // 정렬된 배열 출력
    }
    printf("\n");

    return 0;
}
```
